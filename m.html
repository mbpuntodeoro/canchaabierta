import { getFunctions, httpsCallable } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-functions.js"; const functions = getFunctions(); const crearOrganizador = httpsCallable(functions, 'crearOrganizador'); window.altaPro = async () => { const hubId = document.getElementById('new-org-id').value; const nombreOrg = document.getElementById('new-org-nombre').value; const email = document.getElementById('new-org-email').value; const password = document.getElementById('new-org-pass').value; const btn = document.getElementById('btn-crear'); btn.innerText = "⏳ Creando en Google Cloud..."; btn.disabled = true; try { const result = await crearOrganizador({ hubId, nombreOrg, email, password }); console.log("Éxito:", result.data); alert("¡Cliente creado con éxito! Ya puede loguearse con ese mail."); location.reload(); } catch (error) { console.error(error); alert("Error al crear: " + error.message); btn.innerText = "CREAR CUENTA Y VINCULAR HUB"; btn.disabled = false; } };
